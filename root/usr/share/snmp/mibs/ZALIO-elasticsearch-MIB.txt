zalEsIndicesEntry OBJECT-TYPE
    SYNTAX     ZalEsIndicesEntry
    MAX-ACCESS not-accessible
    STATUS     current
    DESCRIPTION
        "A collection of cluster relevant information
         describing its state"
    INDEX { zalEsIndicesIndex }

}
